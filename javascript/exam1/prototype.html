• Array.join()

    배열의 모든 요소를 연결해 하나의 문자열로 만듦

    					var ex1Array = ['apple', 'orange', 'grape'];
						var myVar1 = ex1Array.join(); //myVar1에 'apple,orange,grape' 대입
						var myVar2 = ex1Array.join(''); //myVar2에 'appleorangegrape' 대입
						var myVar3 = ex1Array.join(', '); //myVar3에 'apple, orange, grape' 대입
						var myVar4 = ex1Array.join('-'); //myVar4에 'apple-orange-grape' 대입
						var myVar5 = ex1Array.join('+'); //myVar5에 'apple+orange+grape' 대입
  • Array.reverse()

    배열의 순서 반전. 첫 번째 요소는 마지막 요소, 마지막 요소는 첫 번째 요소로 됨

    					var ex2Array = ['one', 'two', 'three'];
						console.log(ex2Array); //["one", "two", "three"]

						ex2Array.reverse();
						console.log(ex2Array); //["three", "two", "one"]
  • Array.sort()

    배열의 요소를 적절한 위치에 정렬한 후 그 배열을 반환.

    					var day = ['sun', 'tue', 'mon', 'sat', 'wed', 'thu', 'fri'];
						day.sort();
						console.log(day); //["fri", "mon", "sat", "sun", "thu", "tue", "wed"]
  • Array.concat()

    인자로 주어진 배열이나 값들을 기존 배열에 합쳐서 새 배열을 반환

    					var array1 = ['a', 'b', 'c'];
						var array2 = ['d', 'e', 'f'];
						var array3 = array1.concat(array2);
						console.log(array3); //["a", "b", "c", "d", "e", "f"]
  • Array.slice()

    배열의 begin부터 end까지(end미포함)의 배열 내용 반환. 원본 배열은 바뀌지 않음

    					var fruits = ['Banana', 'Orange', 'Lemon', 'Apple', 'Mango']
						var citrus = fruits.slice(1, 3)
						console.log(fruits); //["Banana", "Orange", "Lemon", "Apple", "Mango"]
						console.log(citrus); //["Orange","Lemon"]
  • Array.splice()

    배열의 기존 요소를 삭제 또는 교체하거나 새 요소를 추가하여 배열의 내용 변경
    array.splice(start[, deleteCount[, item1[, item2[, ...]]]]) 

    					ex1) 하나도 제거하지 않고, 2번 인덱스에 "drum" 추가
						var myFish = ['angel', 'clown', 'mandarin', 'sturgeon'];
						var removed = myFish.splice(2, 0, 'drum');
						console.log(myFish); //["angel", "clown", "drum", "mandarin", "sturgeon"]
						console.log(removed); //[]

						ex2) 3번 인덱스에서 한 개 요소 제거
						var myFish = ['angel', 'clown', 'drum', 'mandarin', 'sturgeon'];
						var removed = myFish.splice(3, 1);
						console.log(removed); //["mandarin"]
						console.log(myFish); //["angel", "clown", "drum", "sturgeon"]

						ex3) 0번 인덱스에서 두 개 요소 제거하고 "parrot", "anemone", "blue" 추가
						var myFish = ['angel', 'clown', 'trumpet', 'sturgeon'];
						var removed = myFish.splice(0, 2, 'parrot', 'anemone', 'blue');
						console.log(myFish); //["parrot", "anemone", "blue", "trumpet", "sturgeon"]
						console.log(removed); //["angel", "clown"]

						ex4) 2번 인덱스에서 두 개 요소 제거
						var myFish = ['parrot', 'anemone', 'blue', 'trumpet', 'sturgeon'];
						var removed = myFish.splice(myFish.length - 3, 2);
						console.log(myFish); //["parrot", "anemone", "sturgeon"]
						console.log(removed); //["blue", "trumpet"]

						ex5) -2번 인덱스에서 한 개 요소 제거
						var myFish = ['angel', 'clown', 'mandarin', 'sturgeon'];
						var removed = myFish.splice(-2, 1);
						console.log(myFish); //["angel", "clown", "sturgeon"]
						console.log(removed); //["mandarin"]

						ex6) 2번 인덱스를 포함해서 이후의 모든 요소 제거
						var myFish = ['angel', 'clown', 'mandarin', 'sturgeon'];
						var removed = myFish.splice(2);
						console.log(myFish); //["angel", "clown"]
						console.log(removed); //["mandarin", "sturgeon"]
  • Array.push()

    배열의 끝에 하나 이상의 요소를 추가하고, 배열의 새로운 길이를 반환

    					ex1)배열에 엘리먼트 추가
						var sports = ['축구', '야구'];
						var total = sports.push('미식축구', '수영');
						console.log(sports); //["축구", "야구", "미식축구", "수영"]
						console.log(total); //4

						ex2)두개의 배열 합치기(모든 엘리먼트를 push하기위해 apply()를 사용)
						var vegetables = ['설탕당근', '감자'];
						var moreVegs = ['셀러리', '홍당무'];
						Array.prototype.push.apply(vegetables, moreVegs);
						console.log(vegetables); //['설탕당근', '감자', '셀러리', '홍당무']
  • Array.pop()

    배열에서 마지막 요소를 제거하고 그 요소를 반환

    					var myFish = ['angel', 'clown', 'mandarin', 'sturgeon'];
						var popped = myFish.pop();
						console.log(myFish); //["angel", "clown", "mandarin"]
						console.log(popped); //sturgeon
  • Array.shift()

    배열에서 첫 번째 요소를 제거하고, 제거된 요소를 반환. 배열의 길이를 변하게 함

    					var array1 = [1, 2, 3];
						var firstElement = array1.shift();
						console.log(array1); //[2, 3]
						console.log(firstElement); //1
  • Array.unshift()

    새로운 요소를 배열의 맨 앞쪽에 추가하고, 새로운 길이를 반환

    					var arr = [1, 2];
					console.log(arr.unshift(0)); //3
					console.log(arr); //[0, 1, 2]

					console.log(arr.unshift([-3])); //4 
					console.log(arr); //[[-3], 0, 1, 2]
  • Array.toString()

    지정된 배열 및 그 요소를 나타내는 문자열 반환

    					var monthNames = ['Jan', 'Feb', 'Mar', 'Apr'];
						var myVar = monthNames.toString();
						console.log(myVar); //'Jan,Feb,Mar,Apr'
  • Array.indexOf()

    배열에서 지정된 요소를 찾을 수 있는 첫 번째 인덱스를 반환하고 존재하지 않으면 -1을 반환
    arr.indexOf(searchElement[, fromIndex]) 

    					var beasts = ['ant', 'bison', 'camel', 'duck', 'bison'];
						console.log(beasts.indexOf('bison')); //1
						console.log(beasts.indexOf('bison', 2)); //start from index 2 , output:4
						console.log(beasts.indexOf('giraffe')); //-1
  • String.replace()

    어떤 패턴에 일치하는 일부 또는 모든 부분이 교체된 새로운 문자열 반환.
    패턴은 문자열이나 정규식이 될 수 있으며, 교체 문자열은 문자열이나 모든 매치에 대해서 호출된 함수일 수 있음.

    					var p = 'The quick brown fox jumps over the lazy dog. If the dog reacted, was it really lazy?';
						console.log(p.replace('dog', 'monkey')); 
						//The quick brown fox jumps over the lazy monkey. If the dog reacted, was it really lazy?

						var regex = /dog/i;
						console.log(p.replace(regex, 'ferret')); 
						//The quick brown fox jumps over the lazy ferret. If the dog reacted, was it really lazy?
  • String.slice()

    문자열의 일부를 추출하면서 새로운 문자열을 반환
    slice()는 endIndex를 포함하지 않고 추출함
    str.slice(1, 4)는 두 번째 문자부터 네 번째 문자까지 추출(1, 2, 3 인덱스 문자)
    str.slice(2, -1)는 세 번째 문자부터 문자열의 마지막에서 두 번째 문자까지 추출 

    					var str1 = 'The morning is upon us.', // the length of str1 is 23.
						str2 = str1.slice(1, 8),
						str3 = str1.slice(4, -2),
						str4 = str1.slice(12),
						str5 = str1.slice(30);
						console.log(str2); //he morn
						console.log(str3); //morning is upon u
						console.log(str4); //is upon us.
						console.log(str5); //""

						var str = 'The morning is upon us.';
						str.slice(-3);     //'us.'
						str.slice(-3, -1); //'us'
						str.slice(0, -1);  //'The morning is upon us'

  • String.search()

  • String.match()

    문자열이 정규표현식에 맞는 문자열을 찾아서 배열 객체로 반환
    일치하는 것이 없으면 null 반환

    					var str = 'red is impressive.'

						console.log(str2); //red가 있으므로 red를 출력

						var test  = 'love you. love me. love everything!'

						if (!location.href.match('/en')) { //현재 url 주소에 '/en'이 포함되어 있는지 match() 함수를 사용하여 확인
							console.log('English page!'); //현재 접속자 url에 /en이 포함되어 있다면 'English page!'
						} else {
							console.log('Korean page!'); //아니면 'Korean page!'
						}
  • String.trim()

    					var greeting = '   Hello world!   ';

  • String.indexOf()

    호출한 String 객체에서 주어진 값과 일치하는 첫 번째 인덱스를 반환, 일치하는 값이 없으면 -1을 반환

    					'Blue Whale'.indexOf('Blue');     // returns  0
						'Blue Whale'.indexOf('Blute');    // returns -1
						'Blue Whale'.indexOf('Whale', 0); // returns  5
						'Blue Whale'.indexOf('Whale', 5); // returns  5
						'Blue Whale'.indexOf('Whale', 7); // returns -1
						'Blue Whale'.indexOf('');         // returns  0
						'Blue Whale'.indexOf('', 9);      // returns  9
						'Blue Whale'.indexOf('', 10);     // returns 10
						'Blue Whale'.indexOf('', 11);     // returns 10
  • Object.hasOwnProperty()

    객체가 특정 프로퍼티를 가지고 있는지를 나타내는 불리언 값을 반환

    					ex1)프로퍼티의 존재 여부 테스트
						o = new Object();
						o.prop = 'exists';
						function change() {
							o.newprop = o.prop;
							delete o.prop;
						}
						o.hasOwnProperty('prop'); //true
						change();
						o.hasOwnProperty('prop'); //false

						ex2)직접 프로퍼티와 상속된 프로퍼티의 비교
						a = new Object();
						a.prop = 'exists';
						a.hasOwnProperty('prop'); // true
						a.hasOwnProperty('toString'); // false
						a.hasOwnProperty('hasOwnProperty'); //false

						ex3)객체의 프로퍼티들을 순환
						var buz = {
							fog: 'stack'
						};
						  
						for (var name in buz) {
							if (buz.hasOwnProperty(name)) {
							  console.log('this is fog (' + name + ') for sure. Value: ' + buz[name]); //this is fog (fog) for sure. Value: stack
							}
							else {
							  console.log(name); // toString or something else
							}
						}

						ex4)프로퍼티의 명칭으로서 hasOwnProperty 사용하기
